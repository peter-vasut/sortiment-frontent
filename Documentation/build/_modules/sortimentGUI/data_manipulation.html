    :param obj: user or food
    :return: list of strings
    """

    res = list()
    if "name" in dir(obj):
        if obj.name is not None:
            res.append(obj.name)

        if obj.nick is not None:

[docs]def normalize_string(s, lowercase=True, special=True):
    output = ''
    if special:
        s = unicodedata.normalize('NFKD', s)
        for c in s:
            if not unicodedata.combining(c):
                if c in string.ascii_letters:
                    output += c
    else:
        output = s
    if lowercase:
        output = output.lower()
    return output




[docs]def format_money(number, separator=",", currency=default_currency):
    return ("-" if number < 0 else "") + str(abs(number) // 100) + separator + "{:02d}".format(
        abs(number) % 100) + currency

[docs]def price_string_to_int(s):
    seppos = s.find(".")
    if seppos < 0:
        seppos = s.find(",")
    if seppos >= 0:
        r = s[:seppos] + s[seppos + 1:seppos + 3]
    else:
        r = int(s) * 100
    return int(r)
